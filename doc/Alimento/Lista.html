def extraerPrimero
	if @size < 2
		if @head.nil?
			false
		else
			@head = nil
			@tail = nil
			@size = 0
			true
		end
	else
		@head = @head[:next]
		@head[:prev] = nil
		@size -= 1
		true
	end

end

def insertar(value)
	if value.respond_to? :each
		value.each { |x| 
			insertarValor(x)
		}
	else
		insertarValor(value)
	end

	true
end

def insertarValor(value)
	if !@tail.nil?
                              @nodo = @@Nodo.new(value, @tail, nil)
                              @tail[:next] = @nodo
                              @tail = @nodo
	else
		@tail = @head = @@Nodo.new(value, @tail, nil)
	end
	
	@size += 1

	true
end
